新版sodimas的wifi模块的控制
在/usr/module 目录下
有 udhcpd.conf 和 hostapd.conf 分别能够配置wifi的IP等信息


[root@IoT-A6G2C module]# cat udhcpd.conf 
start 192.168.5.10
end 192.168.5.30 --->支持21个，可以改成只支持一个的

interface wlan0

option subnet 255.255.255.0
option router 192.168.5.1
option domain local
option lease 864000
#echo "1" > /proc/sys/net/ipv4/ip_forward
[root@IoT-A6G2C module]#
